    :speak_no_evil: @iterable-iterator/filter

    Iterable filtering functions for JavaScript. See docs.

    :warning: Depending on your environment, the code may require regeneratorRuntime to be defined, for instance by importing regenerator-runtime/runtime.

    import {count} from '@iterable-iterator/count';
import {filter, filterfalse} from '@iterable-iterator/filter';
import {divisible} from '@aureooms/js-predicate';

filter( divisible( 2 ) , count( 0 , 1 ) ) ; // 0 2 4 8 ...
filterfalse( divisible( 2 ) , count( 0 , 1 ) ) ; // 1 3 5 7 ...
